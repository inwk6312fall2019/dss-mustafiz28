def has_duplicate(list1):
  list2 = []
  for item in list1:
    if item not in list2:
      list2.append(item)
    else:
      list2[item] +=1
      if list2[item]>=2:
        return True
    return False

list3 = [1,2,3,2,4,5]
list4 = ["a","b","c","d","b"] 
list5 = [11,22,33,44,45]
print(has_duplicate(list3))
print(has_duplicate(list4))
print(has_duplicate(list5))
